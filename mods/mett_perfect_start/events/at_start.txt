namespace = at_start

event = {
	id = at_start.00
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { is_ai = no }
			country_event = { id = at_start.01 }
		}
	}
}

country_event = {
	id = at_start.01
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		every_owned_leader = {
			limit = { leader_class = governor }
			remove_governor_traits = yes
			if = {
				limit = { NOT = { has_authority = auth_machine_intelligence } }
				add_governor_traits_bio = yes
			}
			else = {
				add_governor_traits_machine = yes
			}
		}

		if = {
			limit = { NOT = { has_ethic = ethic_gestalt_consciousness } }
			every_owned_leader = {
				limit = { leader_class = ruler }
				remove_ruler_traits = yes
				add_ruler_traits = yes
			}
		}
	}
}
